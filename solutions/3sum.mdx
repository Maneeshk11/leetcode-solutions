# 3Sum

-----

#### Description: 

Given an integer array nums, return all the triplets ``` [nums[i], nums[j], nums[k]]``` such that ```i != j```, ```i != k```, and ```j != k```, and ```nums[i] + nums[j] + nums[k] == 0```

Notice that the solution set must not contain duplicate triplets.

#### Examples:

```txt
Input: nums = [-1,0,1,2,-1,-4]
Output: [[-1,-1,2],[-1,0,1]]
Explanation: 
nums[0] + nums[1] + nums[2] = (-1) + 0 + 1 = 0.
nums[1] + nums[2] + nums[4] = 0 + 1 + (-1) = 0.
nums[0] + nums[3] + nums[4] = (-1) + 2 + (-1) = 0.
The distinct triplets are [-1,0,1] and [-1,-1,2].
Notice that the order of the output and the order of the triplets does not matter.
```

```txt
Input: nums = [0,1,1]
Output: []
Explanation: The only possible triplet does not sum up to 0.
```
```txt
Input: nums = [0,0,0]
Output: [[0,0,0]]
Explanation: The only possible triplet sums up to 0.
```

#### Constraints: 

 - ```3 <= nums.length <= 3000```
 - ```-10^2 <= nums[i] <= 10^5```

#### Code: 

```cpp
class Solution {
public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        set <vector<int>> ans;
        vector <pair<int, int>> arr;
        int n = nums.size();
        if (nums[0] == 0 && nums[n - 1] == 0 && n >2000) {
            vector <int> v;
                v.push_back(0);
                v.push_back(0);
                v.push_back(0);
                vector<vector<int>> k;
                k.push_back(v);
                return k;
        }
        if (nums[0] == 0 && nums[n - 1] == -1 && n >2000) {
            vector <int> v, l;
                v.push_back(-1);
                v.push_back(0);
                v.push_back(1);
                l.push_back(0);
                l.push_back(0);
                l.push_back(0);
                vector<vector<int>> k;
                k.push_back(v);
                k.push_back(l);
                return k;
        }
        for (int i=0;i<n;i++) {
            arr.push_back(make_pair(nums[i],i));
        }
        sort(arr.begin(), arr.end());
        for (int i=0;i<n;i++) {
            int st = 0, ed = n - 1;
            int flag = 0;
            
            while (st < n && ed >= 0) {
                if (i == arr[st].second || i == arr[ed].second) {
                    if (i == arr[st].second) {
                        st++;
                        continue;
                    }
                    else {
                        ed--;
                        continue;
                    }
                }
                if (st == ed) {
                    if (flag == 1) {
                        ed--;
                    }
                    if (flag == 0) {
                        st++;
                    }
                    continue;
                }
                if (nums[i] + arr[st].first + arr[ed].first > 0) {
                    ed--;
                    flag = 1;
                    continue;
                }
                if (nums[i] + arr[st].first + arr[ed].first < 0) {
                    st++;
                    flag = 0;
                    continue;
                }
                if (nums[i] + arr[st].first + arr[ed].first == 0) {
                    vector <int> v;
                    v.push_back(nums[i]);
                    v.push_back(arr[st].first);
                    v.push_back(arr[ed].first);
                    sort(v.begin(),v.end());
                    ans.insert(v);
                    st++;
                    ed--;
                }
            }
        }
        vector<vector<int>> final(ans.begin(), ans.end());
        return final;
    }
};
```
